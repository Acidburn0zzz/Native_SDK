56 inline float growFloat(const uint32_t& value, const uint8_t& mantissaBits, const uint8_t& exponentBits, const uint8_t& exponentBias, const bool hasSign /*=true*/)
57 {
58  // Union to avoid messy casting.
59  union
60  {
61  float fVal;
62  uint32_t uiVal;
63  } intToFloat;
64 
65  // Float 32 constants.
66  static const uint8_t f32expBias = 127;
67  static const uint8_t f32signBits = 1;
68  static const uint8_t f32expBits = 8;
69  static const uint8_t f32manBits = 23;
70  static const uint8_t f32totBits = 32;
71 
72  // Get the # of sign bits (always 1 or 0)
73  uint8_t signBits = (hasSign ? 1 : 0);
74 
75  // Work out the total bits
76  uint8_t totalBits = mantissaBits + exponentBits + signBits;
77 
78  // Generate the masks for each bit.
79  uint32_t signMask = ((1 << signBits) - 1) << (exponentBits + mantissaBits);
80  uint32_t expoMask = ((1 << exponentBits) - 1) << mantissaBits;
81  uint32_t mantMask = ((1 << mantissaBits) - 1);
82 
83  // Get original exponent.
84  uint32_t originalExponent = ((value & expoMask) >> (mantissaBits));
85  bool originalIsNaN = (originalExponent == (uint32_t)((1 << exponentBits) - 1));
86 
87  // Bit shift and edit until
88  uint32_t uSign = (value & signMask) << ((f32totBits - f32signBits) - (totalBits - signBits));
89  uint32_t uExponent = (originalExponent - exponentBias + f32expBias) << f32manBits;
90  uint32_t uMantissa = (value & mantMask) << (f32manBits - mantissaBits);
91 
92  // Make sure NaN/Infinity is preserved if original was NaN/Infinity.
93  if (originalIsNaN)
94  uExponent = ((1 << f32expBits) - 1) << f32manBits;
95 
96  // Assign the expanded values to the union
97  intToFloat.uiVal = uSign | uExponent | uMantissa;
98 
99  // Return the expanded float
100  return intToFloat.fVal;
101 }

135 inline void convertXYZToCubeUV(float x, float y, float z, CubeFace& face, float& u, float& v)
136 {
137  float absX = fabs(x);
138  float absY = fabs(y);
139  float absZ = fabs(z);
140 
141  if (absX == 0.f && absY == 0.f && absZ == 0.f)
142  {
143  throw new std::runtime_error("Cannot convert the zero vector to a cubemap sample");
144  }
145 
146  int isXPositive = x > 0 ? 1 : 0;
147  int isYPositive = y > 0 ? 1 : 0;
148  int isZPositive = z > 0 ? 1 : 0;
149 
150  float maxAxis = 0.0f;
151  float uc = 0.0f;
152  float vc = 0.0f;
153 
154  // POSITIVE X
155  if (isXPositive && absX >= absY && absX >= absZ)
156  {
157  // u (0 to 1) goes from +z to -z
158  // v (0 to 1) goes from -y to +y
159  maxAxis = absX;
160  uc = -z;
161  vc = -y;
162  face = CubeFace::PositiveX;
163  }
164  // NEGATIVE X
165  if (!isXPositive && absX >= absY && absX >= absZ)
166  {
167  // u (0 to 1) goes from -z to +z
168  // v (0 to 1) goes from -y to +y
169  maxAxis = absX;
170  uc = z;
171  vc = -y;
172  face = CubeFace::NegativeX;
173  }
174  // POSITIVE Y
175  if (isYPositive && absY >= absX && absY >= absZ)
176  {
177  // u (0 to 1) goes from -x to +x
178  // v (0 to 1) goes from +z to -z
179  maxAxis = absY;
180  uc = x;
181  vc = z;
182  face = CubeFace::PositiveY;
183  }
184  // NEGATIVE Y
185  if (!isYPositive && absY >= absX && absY >= absZ)
186  {
187  // u (0 to 1) goes from -x to +x
188  // v (0 to 1) goes from -z to +z
189  maxAxis = absY;
190  uc = x;
191  vc = +z;
192  face = CubeFace::NegativeY;
193  }
194  // POSITIVE Z
195  if (isZPositive && absZ >= absX && absZ >= absY)
196  {
197  // u (0 to 1) goes from -x to +x
198  // v (0 to 1) goes from -y to +y
199  maxAxis = absZ;
200  uc = x;
201  vc = -y;
202  face = CubeFace::PositiveZ;
203  }
204  // NEGATIVE Z
205  if (!isZPositive && absZ >= absX && absZ >= absY)
206  {
207  // u (0 to 1) goes from +x to -x
208  // v (0 to 1) goes from -y to +y
209  maxAxis = absZ;
210  uc = -x;
211  vc = -y;
212  face = CubeFace::NegativeZ;
213  }
214 
215  // Convert range from -1 to 1 to 0 to 1
216  u = 0.5f * (uc / maxAxis + 1.0f);
217  v = 0.5f * (vc / maxAxis + 1.0f);
218 }
