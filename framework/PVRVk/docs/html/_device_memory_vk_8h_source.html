176  {
177  // allocate the memory
178  VkMemoryAllocateInfo memAllocInfo = {};
179  memAllocInfo.sType = static_cast<VkStructureType>(StructureType::e_MEMORY_ALLOCATE_INFO);
180  memAllocInfo.pNext = nullptr;
181  memAllocInfo.allocationSize = allocationInfo.getAllocationSize();
182  memAllocInfo.memoryTypeIndex = allocationInfo.getMemoryTypeIndex();
183 
184  // handle extension
185  VkExportMemoryAllocateInfoKHR memAllocateInfoKHR = {};
186  if (allocationInfo.getExportMemoryAllocateInfoKHR().handleTypes != ExternalMemoryHandleTypeFlags::e_NONE)
187  {
188  memAllocateInfoKHR.sType = static_cast<VkStructureType>(StructureType::e_EXPORT_MEMORY_ALLOCATE_INFO_KHR);
189  memAllocateInfoKHR.handleTypes = static_cast<VkExternalMemoryHandleTypeFlags>(allocationInfo.getExportMemoryAllocateInfoKHR().handleTypes);
190  memAllocInfo.pNext = &memAllocateInfoKHR;
191  }
192 
193  if (memAllocInfo.memoryTypeIndex == static_cast<uint32_t>(-1))
194  {
195  throw ErrorValidationFailedEXT("Device Memory allocation failed: Could not get a Memory Type Index for the specified combination specified memory bits, properties and "
196  "flags");
197  }
198  vkThrowIfFailed(getDevice()->getVkBindings().vkAllocateMemory(device->getVkHandle(), &memAllocInfo, nullptr, &outMemory), "Failed to allocate device memory");
